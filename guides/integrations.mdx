## How the Stream Works

`/movie` and `/tv` are **Server-Sent Events** — not JSON endpoints. Each response is a stream of `data:` lines. There are three event types:

| Event | When | Contains |
|-------|------|----------|
| `meta` | Immediately, before any provider resolves | TMDB metadata + all subtitle tracks |
| `source` | Once per working provider, as each resolves | Proxied HLS URL + provider label |
| `done` | After all providers finish or time out | Total working source count |

Start playback on the **first `source` event** — don't wait for `done`.

## Minimal HTML Player

```html
<!DOCTYPE html>
<html>
<head>
  <script src="https://cdn.jsdelivr.net/npm/hls.js@latest/dist/hls.min.js"></script>
</head>
<body>
  <video id="player" controls style="width: 100%; max-width: 900px;"></video>
  <div id="status">Loading…</div>

  <script>
    const BASE   = 'https://missourimonster-vyla.hf.space';
    const video  = document.getElementById('player');
    const status = document.getElementById('status');

    async function play(tmdbId) {
      const res     = await fetch(`${BASE}/movie?id=${tmdbId}`);
      const reader  = res.body.getReader();
      const decoder = new TextDecoder();
      let buffer    = '';
      let started   = false;

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop();

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue;
          const event = JSON.parse(line.slice(6));

          if (event.type === 'source' && !started) {
            started = true;
            status.textContent = `Playing via ${event.source.label}`;

            if (!Hls.isSupported()) { video.src = event.source.url; continue; }

            const hls = new Hls();
            hls.loadSource(event.source.url);
            hls.attachMedia(video);
            hls.on(Hls.Events.MANIFEST_PARSED, () => video.play());
          }

          if (event.type === 'done' && !started) {
            status.textContent = 'No sources available.';
          }
        }
      }
    }

    play(550);
  </script>
</body>
</html>
```

## HTML Player with Source Switching + Subtitles

```html
<!DOCTYPE html>
<html>
<head>
  <script src="https://cdn.jsdelivr.net/npm/hls.js@latest/dist/hls.min.js"></script>
  <style>
    body    { font-family: sans-serif; background: #111; color: #fff; padding: 1rem; }
    video   { width: 100%; max-width: 900px; display: block; }
    #sources { display: flex; gap: 0.5rem; flex-wrap: wrap; margin-top: 0.5rem; }
    button  { padding: 0.4rem 0.8rem; background: #333; color: #fff; border: none;
              border-radius: 4px; cursor: pointer; }
    button.active { background: #fff; color: #111; }
    #status { margin-top: 0.5rem; font-size: 0.85rem; color: #aaa; }
  </style>
</head>
<body>
  <video id="player" controls></video>
  <div id="sources"></div>
  <div id="status">Connecting…</div>

  <script>
    const BASE      = 'https://missourimonster-vyla.hf.space';
    const video     = document.getElementById('player');
    const statusEl  = document.getElementById('status');
    const sourcesEl = document.getElementById('sources');
    let hls;

    function switchSource(url, label) {
      hls?.destroy();
      hls = new Hls();
      statusEl.textContent = `Loading ${label}…`;
      hls.on(Hls.Events.MANIFEST_PARSED, () => { statusEl.textContent = `Playing: ${label}`; video.play(); });
      hls.on(Hls.Events.ERROR, (_, err) => { if (err.fatal) statusEl.textContent = 'Source error — try another.'; });
      hls.loadSource(url);
      hls.attachMedia(video);
    }

    async function loadMovie(tmdbId) {
      const res     = await fetch(`${BASE}/movie?id=${tmdbId}`);
      const reader  = res.body.getReader();
      const decoder = new TextDecoder();
      let buffer    = '';
      let started   = false;

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n');
        buffer = lines.pop();

        for (const line of lines) {
          if (!line.startsWith('data: ')) continue;
          const event = JSON.parse(line.slice(6));

          if (event.type === 'meta') {
            statusEl.textContent = `Loading ${event.meta?.title ?? ''}…`;
            event.subtitles?.forEach((sub, i) => {
              const track = Object.assign(document.createElement('track'), {
                kind: 'subtitles', label: sub.label, src: sub.file,
              });
              if (i === 0) track.default = true;
              video.appendChild(track);
            });
          }

          if (event.type === 'source') {
            const { url, label } = event.source;
            const btn = document.createElement('button');
            btn.textContent = label;
            if (!started) btn.classList.add('active');
            btn.onclick = () => {
              sourcesEl.querySelectorAll('button').forEach(b => b.classList.remove('active'));
              btn.classList.add('active');
              switchSource(url, label);
            };
            sourcesEl.appendChild(btn);

            if (!started) { started = true; switchSource(url, label); }
          }

          if (event.type === 'done' && !started) {
            statusEl.textContent = 'No sources available.';
          }
        }
      }
    }

    loadMovie(550);
  </script>
</body>
</html>
```

## Quality Level Selection

HLS.js exposes quality levels from the manifest. Present them to the user after the first source loads:

```javascript
function setupQualitySelector(hls, selectEl) {
  hls.on(Hls.Events.MANIFEST_PARSED, (_, data) => {
    selectEl.innerHTML = '<option value="-1">Auto</option>';
    data.levels.forEach((level, i) => {
      const opt = document.createElement('option');
      opt.value = i;
      opt.textContent = `${level.height}p (${Math.round(level.bitrate / 1000)}kbps)`;
      selectEl.appendChild(opt);
    });
    selectEl.onchange = () => { hls.currentLevel = parseInt(selectEl.value); };
  });

  hls.on(Hls.Events.LEVEL_SWITCHED, (_, data) => {
    selectEl.value = hls.autoLevelEnabled ? '-1' : data.level;
  });
}
```

## React Hook

A reusable hook that handles SSE streaming, source queuing, and error state:

```typescript
// hooks/useStream.ts
import { useState, useEffect, useRef } from 'react';

const BASE = 'https://missourimonster-vyla.hf.space';

interface Source   { source: string; label: string; url: string; }
interface Subtitle { label: string; file: string; type: string; source: string; }

interface StreamState {
  sources:   Source[];
  subtitles: Subtitle[];
  meta:      Record<string, unknown> | null;
  loading:   boolean;
  done:      boolean;
  error:     string | null;
}

interface StreamParams {
  tmdbId:   number | null;
  season?:  number | null;
  episode?: number | null;
}

export function useStream({ tmdbId, season, episode }: StreamParams): StreamState {
  const [state, setState] = useState<StreamState>({
    sources: [], subtitles: [], meta: null, loading: false, done: false, error: null,
  });
  const abortRef = useRef<AbortController | null>(null);

  useEffect(() => {
    if (!tmdbId) return;

    abortRef.current?.abort();
    const controller = new AbortController();
    abortRef.current = controller;

    const url = season && episode
      ? `${BASE}/tv?id=${tmdbId}&season=${season}&episode=${episode}`
      : `${BASE}/movie?id=${tmdbId}`;

    setState({ sources: [], subtitles: [], meta: null, loading: true, done: false, error: null });

    (async () => {
      try {
        const res     = await fetch(url, { signal: controller.signal });
        const reader  = res.body!.getReader();
        const decoder = new TextDecoder();
        let buffer    = '';

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          buffer += decoder.decode(value, { stream: true });
          const lines = buffer.split('\n');
          buffer = lines.pop()!;

          for (const line of lines) {
            if (!line.startsWith('data: ')) continue;
            const event = JSON.parse(line.slice(6));

            if (event.type === 'meta') {
              setState(s => ({ ...s, meta: event.meta ?? null, subtitles: event.subtitles ?? [] }));
            }
            if (event.type === 'source') {
              setState(s => ({ ...s, sources: [...s.sources, event.source] }));
            }
            if (event.type === 'done') {
              setState(s => ({ ...s, loading: false, done: true }));
            }
          }
        }
      } catch (err: any) {
        if (err.name !== 'AbortError') {
          setState(s => ({ ...s, loading: false, error: err.message }));
        }
      }
    })();

    return () => controller.abort();
  }, [tmdbId, season, episode]);

  return state;
}
```

## React Player Component

A full-featured player using the hook above — starts on first source, queues the rest as fallbacks:

```tsx
// components/StreamPlayer.tsx
'use client';
import { useEffect, useRef, useState } from 'react';
import Hls from 'hls.js';
import { useStream } from '@/hooks/useStream';

interface StreamPlayerProps {
  tmdbId:   number;
  season?:  number;
  episode?: number;
}

export function StreamPlayer({ tmdbId, season, episode }: StreamPlayerProps) {
  const videoRef   = useRef<HTMLVideoElement>(null);
  const hlsRef     = useRef<Hls | null>(null);
  const queueRef   = useRef<string[]>([]);
  const startedRef = useRef(false);

  const { sources, subtitles, meta, loading, done, error } = useStream({ tmdbId, season, episode });
  const [activeLabel, setActiveLabel] = useState<string>('');
  const [levels, setLevels] = useState<Hls.Level[]>([]);
  const [level, setLevel]   = useState(-1);

  function playUrl(url: string, label: string) {
    if (!videoRef.current) return;
    hlsRef.current?.destroy();

    if (!Hls.isSupported()) { videoRef.current.src = url; return; }

    const hls = new Hls();
    hlsRef.current = hls;

    hls.on(Hls.Events.MANIFEST_PARSED, (_, data) => {
      setLevels(data.levels);
      setActiveLabel(label);
      videoRef.current?.play().catch(() => {});
    });

    hls.on(Hls.Events.ERROR, (_, err) => {
      if (err.fatal) {
        const next = queueRef.current.shift();
        if (next) playUrl(next, '');
      }
    });

    hls.loadSource(url);
    hls.attachMedia(videoRef.current);
  }

  // Start on first source, queue the rest
  useEffect(() => {
    if (!sources.length) return;
    const latest = sources[sources.length - 1];
    queueRef.current.push(latest.url);
    if (!startedRef.current) {
      startedRef.current = true;
      queueRef.current.shift();
      playUrl(latest.url, latest.label);
    }
  }, [sources]);

  // Sync quality level
  useEffect(() => {
    if (hlsRef.current) hlsRef.current.currentLevel = level;
  }, [level]);

  if (error) return <div>Error: {error}</div>;

  return (
    <div>
      {loading && !sources.length && <div>Connecting to providers…</div>}

      <video ref={videoRef} controls style={{ width: '100%' }}>
        {subtitles.map(sub => (
          <track key={sub.label + sub.source} kind="subtitles" label={sub.label} src={sub.file} />
        ))}
      </video>

      {activeLabel && <p style={{ fontSize: '0.8rem', color: '#aaa' }}>Playing via {activeLabel}</p>}

      {/* Source selector — populated as sources stream in */}
      {sources.length > 1 && (
        <div style={{ display: 'flex', gap: '0.5rem', marginTop: '0.5rem', flexWrap: 'wrap' }}>
          {sources.map(src => (
            <button
              key={src.source}
              onClick={() => playUrl(src.url, src.label)}
              style={{ fontWeight: src.label === activeLabel ? 'bold' : 'normal' }}
            >
              {src.label}
            </button>
          ))}
        </div>
      )}

      {/* Quality selector */}
      {levels.length > 0 && (
        <select value={level} onChange={e => setLevel(Number(e.target.value))}>
          <option value={-1}>Auto</option>
          {levels.map((l, i) => (
            <option key={i} value={i}>{l.height}p</option>
          ))}
        </select>
      )}

      {done && <p style={{ fontSize: '0.75rem', color: '#666' }}>{sources.length} sources loaded</p>}
    </div>
  );
}
```

## Lazy Episode Loading (React)

Don't open a stream until the user picks an episode:

```tsx
'use client';
import { useState } from 'react';
import { StreamPlayer } from '@/components/StreamPlayer';

interface Episode { episode_number: number; name: string; }

export function EpisodeList({
  seriesId, season, episodes,
}: {
  seriesId: number;
  season: number;
  episodes: Episode[];
}) {
  const [selectedEp, setSelectedEp] = useState<number | null>(null);

  return (
    <div style={{ display: 'grid', gridTemplateColumns: '1fr 3fr', gap: '1rem' }}>
      <aside>
        {episodes.map(ep => (
          <button
            key={ep.episode_number}
            onClick={() => setSelectedEp(ep.episode_number)}
            style={{ display: 'block', width: '100%', textAlign: 'left',
                     padding: '0.5rem', fontWeight: selectedEp === ep.episode_number ? 'bold' : 'normal' }}
          >
            E{ep.episode_number} — {ep.name}
          </button>
        ))}
      </aside>
      <section>
        {!selectedEp && <p>Select an episode to watch</p>}
        {selectedEp && (
          <StreamPlayer tmdbId={seriesId} season={season} episode={selectedEp} />
        )}
      </section>
    </div>
  );
}
```

## Safari / iOS

Safari has native HLS support. Detect and handle both paths — the SSE streaming logic is the same; only the player attachment changes:

```javascript
function attachHls(videoEl, sourceUrl) {
  if (Hls.isSupported()) {
    const hls = new Hls();
    hls.loadSource(sourceUrl);
    hls.attachMedia(videoEl);
    return hls;
  }

  if (videoEl.canPlayType('application/vnd.apple.mpegurl')) {
    videoEl.src = sourceUrl;
    return null;
  }

  throw new Error('HLS not supported in this browser');
}
```

## SSR Considerations

<Warning>
  HLS.js must run **client-side only**. Never import it in server components or run it during SSR — it accesses `window` and `document`.

  In Next.js App Router, always add `'use client'` to components that use HLS.js. Use dynamic `import('hls.js')` inside `useEffect` to keep it out of your SSR bundle entirely.
</Warning>

```tsx
// ✅ Correct — dynamic import in useEffect
useEffect(() => {
  import('hls.js').then(({ default: Hls }) => {
    // Safe to use Hls here
  });
}, []);

// ❌ Wrong — top-level import in a component that may SSR
import Hls from 'hls.js'; // will crash on server
```
