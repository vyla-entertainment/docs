## Custom Hook: `useVylaMovie`

A reusable hook that handles fetching, loading state, and error handling for movies.

```typescript
// hooks/useVylaMovie.ts
import { useState, useEffect } from 'react';

const BASE = 'https://missourimonster-vyla-api.hf.space';

interface Source   { source: string; label: string; url: string; }
interface Subtitle { label: string; file: string; }
interface Meta     { [key: string]: unknown; }

interface VylaMovieState {
  sources:   Source[];
  subtitles: Subtitle[];
  meta:      Meta | null;
  loading:   boolean;
  error:     string | null;
}

export function useVylaMovie(tmdbId: number | null): VylaMovieState {
  const [state, setState] = useState<VylaMovieState>({
    sources: [], subtitles: [], meta: null, loading: false, error: null,
  });

  useEffect(() => {
    if (!tmdbId) return;

    setState(s => ({ ...s, loading: true, error: null }));

    fetch(`${BASE}/api/movie?id=${tmdbId}`)
      .then(async r => {
        const data = await r.json();
        if (!r.ok) throw new Error(data.error || `HTTP ${r.status}`);
        setState({
          sources:   data.sources   ?? [],
          subtitles: data.subtitles ?? [],
          meta:      data.meta      ?? null,
          loading:   false,
          error:     null,
        });
      })
      .catch(err => {
        setState(s => ({ ...s, loading: false, error: err.message }));
      });
  }, [tmdbId]);

  return state;
}
```

## Custom Hook: `useVylaTv`

Same pattern for TV episodes:

```typescript
// hooks/useVylaTv.ts
import { useState, useEffect } from 'react';

const BASE = 'https://missourimonster-vyla-api.hf.space';

interface VylaTvParams {
  seriesId: number | null;
  season:   number | null;
  episode:  number | null;
}

export function useVylaTv({ seriesId, season, episode }: VylaTvParams) {
  const [state, setState] = useState({
    sources: [], subtitles: [], meta: null, loading: false, error: null as string | null,
  });

  useEffect(() => {
    if (!seriesId || !season || !episode) return;

    setState(s => ({ ...s, loading: true, error: null }));

    fetch(`${BASE}/api/tv?id=${seriesId}&season=${season}&episode=${episode}`)
      .then(async r => {
        const data = await r.json();
        if (!r.ok) throw new Error(data.error || `HTTP ${r.status}`);
        setState({ sources: data.sources ?? [], subtitles: data.subtitles ?? [], meta: data.meta ?? null, loading: false, error: null });
      })
      .catch(err => {
        setState(s => ({ ...s, loading: false, error: err.message }));
      });
  }, [seriesId, season, episode]);

  return state;
}
```

## Player Component

A clean player component that uses the hooks above:

```tsx
// components/Player.tsx
'use client';
import { useEffect, useRef } from 'react';
import { useVylaMovie } from '@/hooks/useVylaMovie';

const BASE = 'https://missourimonster-vyla-api.hf.space';

export function MoviePlayer({ tmdbId }: { tmdbId: number }) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const hlsRef   = useRef<any>(null);
  const { sources, subtitles, loading, error } = useVylaMovie(tmdbId);

  useEffect(() => {
    if (!sources.length || !videoRef.current) return;

    import('hls.js').then(({ default: Hls }) => {
      if (!Hls.isSupported()) {
        videoRef.current!.src = BASE + sources[0].url;
        return;
      }

      hlsRef.current?.destroy();
      const hls = new Hls();
      hlsRef.current = hls;

      hls.loadSource(BASE + sources[0].url);
      hls.attachMedia(videoRef.current!);
      hls.on(Hls.Events.MANIFEST_PARSED, () => videoRef.current?.play());
    });

    return () => hlsRef.current?.destroy();
  }, [sources]);

  if (loading) return <div className="player-skeleton" />;
  if (error)   return <div className="player-error">{error}</div>;

  return (
    <video ref={videoRef} controls className="w-full rounded-lg">
      {subtitles.map(sub => (
        <track key={sub.label} kind="subtitles" label={sub.label} src={sub.file} />
      ))}
    </video>
  );
}
```

## Next.js App Router Page

A complete movie watch page using the App Router and server-side TMDB metadata:

```typescript
// app/movie/[id]/page.tsx
import { Suspense } from 'react';
import { MoviePlayer } from '@/components/Player';

// Fetch TMDB metadata server-side (optional — you can skip this
// and rely on the meta field in the Vyla API response)
async function getMovieMeta(tmdbId: string) {
  const res = await fetch(
    `https://api.themoviedb.org/3/movie/${tmdbId}?api_key=${process.env.TMDB_API_KEY}`,
    { next: { revalidate: 3600 } } // cache for 1 hour
  );
  return res.ok ? res.json() : null;
}

export default async function MoviePage({
  params,
}: {
  params: { id: string };
}) {
  const meta = await getMovieMeta(params.id);

  return (
    <main className="container mx-auto p-4">
      {meta && (
        <div className="mb-4">
          <h1 className="text-2xl font-bold">{meta.title}</h1>
          <p className="text-sm text-gray-500">{meta.release_date}</p>
        </div>
      )}

      <Suspense fallback={<div className="aspect-video bg-gray-900 animate-pulse rounded-lg" />}>
        <MoviePlayer tmdbId={Number(params.id)} />
      </Suspense>

      {meta?.overview && (
        <p className="mt-4 text-sm text-gray-400">{meta.overview}</p>
      )}
    </main>
  );
}

// Optional: generate static params for popular titles
export async function generateStaticParams() {
  return [{ id: '550' }, { id: '27205' }, { id: '157336' }];
}
```

## Episode List with Lazy Source Loading

Don't fetch sources until the user actually clicks to watch an episode — saves API calls:

```tsx
// components/EpisodeList.tsx
'use client';
import { useState } from 'react';
import { useVylaTv } from '@/hooks/useVylaTv';

interface Episode {
  episode_number: number;
  name: string;
}

export function EpisodeList({
  seriesId,
  season,
  episodes,
}: {
  seriesId: number;
  season: number;
  episodes: Episode[];
}) {
  const [selectedEp, setSelectedEp] = useState<number | null>(null);

  const { sources, subtitles, loading, error } = useVylaTv({
    seriesId,
    season,
    episode: selectedEp,
  });

  return (
    <div className="grid grid-cols-1 gap-4 md:grid-cols-3">
      {/* Episode list */}
      <aside className="md:col-span-1">
        {episodes.map(ep => (
          <button
            key={ep.episode_number}
            onClick={() => setSelectedEp(ep.episode_number)}
            className={`block w-full text-left p-3 rounded ${
              selectedEp === ep.episode_number ? 'bg-blue-600' : 'hover:bg-gray-700'
            }`}
          >
            E{ep.episode_number} — {ep.name}
          </button>
        ))}
      </aside>

      {/* Player */}
      <section className="md:col-span-2">
        {!selectedEp && <p className="text-gray-400">Select an episode</p>}
        {loading && <div className="aspect-video bg-gray-900 animate-pulse rounded-lg" />}
        {error && <p className="text-red-400">{error}</p>}
        {sources.length > 0 && (
          <video controls className="w-full rounded-lg">
            {subtitles.map(s => (
              <track key={s.label} kind="subtitles" label={s.label} src={s.file} />
            ))}
          </video>
        )}
      </section>
    </div>
  );
}
```

## SSR Considerations

<Warning>
  HLS.js must run **client-side only**. Never import it in server components or run it during SSR — it accesses `window` and `document`.

  In Next.js App Router, always add `'use client'` to any component that uses HLS.js or refs to `<video>` elements. Use `import('hls.js')` (dynamic) to keep it out of your SSR bundle entirely.
</Warning>

```tsx
// ✅ Correct — dynamic import in useEffect
useEffect(() => {
  import('hls.js').then(({ default: Hls }) => {
    // Safe to use Hls here
  });
}, []);

// ❌ Wrong — top-level import in a component that may SSR
import Hls from 'hls.js'; // will crash on server
```
