## Basic Playback

```html
<!DOCTYPE html>
<html>
<body>
  <video id="player" controls style="width: 100%; max-width: 900px;"></video>

  <script src="https://cdn.jsdelivr.net/npm/hls.js@latest/dist/hls.min.js"></script>
  <script>
    const BASE = 'https://missourimonster-vyla-api.hf.space';

    async function play(tmdbId) {
      const data = await fetch(`${BASE}/api/movie?id=${tmdbId}`)
        .then(r => r.json());

      if (!data.sources?.length) {
        alert('No sources available');
        return;
      }

      const video = document.getElementById('player');

      if (!Hls.isSupported()) {
        // Safari: native HLS
        video.src = BASE + data.sources[0].url;
        return;
      }

      const hls = new Hls();
      hls.loadSource(BASE + data.sources[0].url);
      hls.attachMedia(video);

      hls.on(Hls.Events.MANIFEST_PARSED, () => video.play());
    }

    play(550); // Fight Club
  </script>
</body>
</html>
```

## Source Switching UI

Build a source picker so users can switch providers if one is buffering:

```html
<video id="player" controls></video>
<div id="sources"></div>

<script>
  const BASE = 'https://missourimonster-vyla-api.hf.space';
  let hls;

  async function loadMovie(tmdbId) {
    const data = await fetch(`${BASE}/api/movie?id=${tmdbId}`).then(r => r.json());

    // Build source buttons
    const container = document.getElementById('sources');
    data.sources.forEach((source, i) => {
      const btn = document.createElement('button');
      btn.textContent = source.label;
      btn.onclick = () => switchSource(source.url, btn, container);
      if (i === 0) btn.classList.add('active');
      container.appendChild(btn);
    });

    // Load first source
    switchSource(data.sources[0].url);

    // Attach subtitles
    const video = document.getElementById('player');
    data.subtitles.forEach(sub => {
      const track = Object.assign(document.createElement('track'), {
        kind: 'subtitles', label: sub.label, src: sub.file,
      });
      video.appendChild(track);
    });
  }

  function switchSource(url, activeBtn, container) {
    hls?.destroy();
    hls = new Hls();

    if (activeBtn && container) {
      container.querySelectorAll('button').forEach(b => b.classList.remove('active'));
      activeBtn.classList.add('active');
    }

    const video = document.getElementById('player');
    hls.loadSource(BASE + url);
    hls.attachMedia(video);
    hls.on(Hls.Events.MANIFEST_PARSED, () => video.play());
  }

  loadMovie(550);
</script>
```

## Quality Level Selection

HLS.js exposes quality levels from the manifest. Present them to the user:

```javascript
let hls;

function setupQualitySelector(hls, selectEl) {
  hls.on(Hls.Events.MANIFEST_PARSED, (_, data) => {
    // Clear existing options
    selectEl.innerHTML = '<option value="-1">Auto</option>';

    data.levels.forEach((level, i) => {
      const opt = document.createElement('option');
      opt.value = i;
      opt.textContent = `${level.height}p (${Math.round(level.bitrate / 1000)}kbps)`;
      selectEl.appendChild(opt);
    });

    selectEl.onchange = () => {
      hls.currentLevel = parseInt(selectEl.value);
    };
  });

  // Show currently playing level
  hls.on(Hls.Events.LEVEL_SWITCHED, (_, data) => {
    if (hls.autoLevelEnabled) {
      selectEl.value = '-1';
    } else {
      selectEl.value = data.level;
    }
  });
}
```

const video = document.getElementById('player');
const spinner = document.getElementById('spinner');

hls.on(Hls.Events.MANIFEST_LOADING, () => {
  spinner.style.display = 'block';
});

hls.on(Hls.Events.MANIFEST_PARSED, () => {
  spinner.style.display = 'none';
});

video.addEventListener('waiting', () => {
  spinner.style.display = 'block';
});

video.addEventListener('playing', () => {
  spinner.style.display = 'none';
});

## Complete React Player Component

A production-ready player component with source switching, subtitle selection, quality control, and error fallback:

```tsx
import { useEffect, useRef, useState, useCallback } from 'react';
import Hls from 'hls.js';

const BASE = 'https://missourimonster-vyla-api.hf.space';

interface Source   { source: string; label: string; url: string; }
interface Subtitle { label: string; file: string; }

interface VylaPlayerProps {
  tmdbId: number;
  isTv?: boolean;
  season?: number;
  episode?: number;
}

export function VylaPlayer({ tmdbId, isTv, season, episode }: VylaPlayerProps) {
  const videoRef  = useRef<HTMLVideoElement>(null);
  const hlsRef    = useRef<Hls | null>(null);

  const [sources,   setSources]   = useState<Source[]>([]);
  const [subtitles, setSubtitles] = useState<Subtitle[]>([]);
  const [active,    setActive]    = useState(0);
  const [loading,   setLoading]   = useState(true);
  const [error,     setError]     = useState<string | null>(null);
  const [levels,    setLevels]    = useState<Hls.Level[]>([]);
  const [level,     setLevel]     = useState(-1);

  // Fetch sources on mount
  useEffect(() => {
    const url = isTv
      ? `${BASE}/api/tv?id=${tmdbId}&season=${season}&episode=${episode}`
      : `${BASE}/api/movie?id=${tmdbId}`;

    fetch(url)
      .then(r => r.json())
      .then(data => {
        if (!data.sources?.length) throw new Error('No sources available');
        setSources(data.sources);
        setSubtitles(data.subtitles ?? []);
        setLoading(false);
      })
      .catch(err => {
        setError(err.message);
        setLoading(false);
      });
  }, [tmdbId, isTv, season, episode]);

  // Load active source
  useEffect(() => {
    if (!sources.length || !videoRef.current) return;

    hlsRef.current?.destroy();

    if (!Hls.isSupported()) {
      videoRef.current.src = BASE + sources[active].url;
      return;
    }

    const hls = new Hls();
    hlsRef.current = hls;

    hls.on(Hls.Events.MANIFEST_PARSED, (_, data) => {
      setLevels(data.levels);
      videoRef.current?.play().catch(() => {});
    });

    hls.on(Hls.Events.ERROR, (_, err) => {
      if (err.fatal && active < sources.length - 1) setActive(i => i + 1);
    });

    hls.loadSource(BASE + sources[active].url);
    hls.attachMedia(videoRef.current);

    return () => hls.destroy();
  }, [sources, active]);

  // Sync quality level
  useEffect(() => {
    if (hlsRef.current) hlsRef.current.currentLevel = level;
  }, [level]);

  if (loading) return <div>Loading sources…</div>;
  if (error)   return <div>Error: {error}</div>;

  return (
    <div>
      <video ref={videoRef} controls style={{ width: '100%' }}>
        {subtitles.map(sub => (
          <track key={sub.label} kind="subtitles" label={sub.label} src={sub.file} />
        ))}
      </video>

      {/* Source selector */}
      <div>
        {sources.map((src, i) => (
          <button
            key={src.source}
            onClick={() => setActive(i)}
            style={{ fontWeight: i === active ? 'bold' : 'normal' }}
          >
            {src.label}
          </button>
        ))}
      </div>

      {/* Quality selector */}
      {levels.length > 0 && (
        <select value={level} onChange={e => setLevel(Number(e.target.value))}>
          <option value={-1}>Auto</option>
          {levels.map((l, i) => (
            <option key={i} value={i}>{l.height}p</option>
          ))}
        </select>
      )}
    </div>
  );
}
```

## Safari / iOS

Safari has native HLS support via `<video>` and doesn't need HLS.js. Detect and handle both:

```javascript
function loadHls(videoEl, url) {
  const fullUrl = 'https://missourimonster-vyla-api.hf.space' + url;

  if (Hls.isSupported()) {
    const hls = new Hls();
    hls.loadSource(fullUrl);
    hls.attachMedia(videoEl);
    return hls;
  }

  if (videoEl.canPlayType('application/vnd.apple.mpegurl')) {
    // Safari native HLS
    videoEl.src = fullUrl;
    return null;
  }

  throw new Error('HLS not supported in this browser');
}
```

<Warning>
  Safari's native HLS player fetches segments directly from the proxied URLs — which works fine since the proxy rewrites all segment paths. However, Safari does not support AES-256-CBC encrypted HLS on some providers. If playback fails on Safari, try a different source.
</Warning>
