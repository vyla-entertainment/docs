## Before You Start

You'll need a TMDB ID for the content you want to stream. Every movie and TV show on [themoviedb.org](https://www.themoviedb.org) has one — it's the number in the URL.

| URL | TMDB ID | Title |
|-----|---------|-------|
| `themoviedb.org/movie/550` | `550` | Fight Club |
| `themoviedb.org/movie/27205` | `27205` | Inception |
| `themoviedb.org/tv/1396` | `1396` | Breaking Bad |
| `themoviedb.org/tv/94997` | `94997` | House of the Dragon |

## Step 1 — Fetch Sources

Call `/api/movie` or `/api/tv`. The API fans out to all configured providers in parallel, verifies every URL live, and returns only the working ones.

<CodeGroup>
```bash Movie
curl https://missourimonster-vyla-api.hf.space/api/movie?id=550
```

```bash TV Episode
curl "https://missourimonster-vyla-api.hf.space/api/tv?id=1396&season=1&episode=1"
```
</CodeGroup>

## Step 2 — Play with HLS.js

The `url` field on each source is a fully-qualified URL — the base is already baked in. Pass it directly to your player with no modification needed.

<Tabs>
  <Tab title="Vanilla JS">
    ```html
    <!DOCTYPE html>
    <html>
    <head>
      <script src="https://cdn.jsdelivr.net/npm/hls.js@latest/dist/hls.min.js"></script>
    </head>
    <body>
      <video id="player" controls style="width:100%"></video>

      <script>
        async function loadMovie(tmdbId) {
          const res  = await fetch(`https://missourimonster-vyla-api.hf.space/api/movie?id=${tmdbId}`);
          const data = await res.json();
          if (!data.sources?.length) return console.error('No sources');

          const video = document.getElementById('player');

          if (!Hls.isSupported()) {
            video.src = data.sources[0].url;
            return;
          }

          const hls = new Hls();
          hls.loadSource(data.sources[0].url);
          hls.attachMedia(video);

          hls.on(Hls.Events.MANIFEST_PARSED, () => {
            data.subtitles.forEach(sub => {
              const track  = document.createElement('track');
              track.kind   = 'subtitles';
              track.label  = sub.label;
              track.src    = sub.file;
              video.appendChild(track);
            });
            video.play();
          });
        }

        loadMovie(550);
      </script>
    </body>
    </html>
    ```
  </Tab>
  <Tab title="React">
    ```tsx
    import { useEffect, useRef } from 'react';
    import Hls from 'hls.js';

    export function Player({ tmdbId }: { tmdbId: number }) {
      const videoRef = useRef<HTMLVideoElement>(null);

      useEffect(() => {
        let hls: Hls;

        async function load() {
          const res  = await fetch(`https://missourimonster-vyla-api.hf.space/api/movie?id=${tmdbId}`);
          const data = await res.json();
          if (!data.sources?.length || !videoRef.current) return;

          if (!Hls.isSupported()) {
            videoRef.current.src = data.sources[0].url;
            return;
          }

          hls = new Hls();
          hls.loadSource(data.sources[0].url);
          hls.attachMedia(videoRef.current);
          hls.on(Hls.Events.MANIFEST_PARSED, () => {
            videoRef.current?.play();
          });
        }

        load();
        return () => hls?.destroy();
      }, [tmdbId]);

      return <video ref={videoRef} controls style={{ width: '100%' }} />;
    }
    ```
  </Tab>

    import requests

    BASE = "https://missourimonster-vyla-api.hf.space"

    def get_sources(tmdb_id: int, season: int = None, episode: int = None):
        if season and episode:
            url = f"{BASE}/api/tv"
            params = {"id": tmdb_id, "season": season, "episode": episode}

        res.raise_for_status()
        data = res.json()

        return {
            "sources": data.get("sources", []),
            "subtitles": data.get("subtitles", []),
        }

    result = get_sources(550)
    for source in result["sources"]:
        print(f"{source['label']}: {source['url']}")

## Step 3 — Handle Fallbacks

Providers can go down at any time. Build a fallback loop so your player automatically tries the next source on failure.

```typescript
async function loadWithFallback(
  videoEl: HTMLVideoElement,
  tmdbId: number,
  season?: number,
  episode?: number
) {
  const endpoint = season && episode
    ? `https://missourimonster-vyla-api.hf.space/api/tv?id=${tmdbId}&season=${season}&episode=${episode}`
    : `https://missourimonster-vyla-api.hf.space/api/movie?id=${tmdbId}`;

  const data = await fetch(endpoint).then(r => r.json());
  if (!data.sources?.length) throw new Error('No sources available');

  const urls: string[] = data.sources.map((s: any) => s.url);
  let index = 0;
  let hls: any;

  function tryNext() {
    if (index >= urls.length) {
      console.error('All sources exhausted');
      return;
    }

    hls?.destroy();
    hls = new Hls();

    hls.on(Hls.Events.ERROR, (_: any, err: any) => {
      if (err.fatal) {
        console.warn(`Source ${index} failed, trying next`);
        index++;
        tryNext();
      }
    });

    hls.loadSource(urls[index]);
    hls.attachMedia(videoEl);
    hls.on(Hls.Events.MANIFEST_PARSED, () => videoEl.play());
  }

  tryNext();
}
```

## Step 4 — Add Subtitles

Subtitles are included automatically in movie and TV responses. You can also fetch them standalone if needed.

```javascript
// Already included in /api/movie and /api/tv responses
const { sources, subtitles } = await fetch(
  'https://missourimonster-vyla-api.hf.space/api/movie?id=550'
).then(r => r.json());

// Or fetch standalone
const subtitles = await fetch(
  'https://missourimonster-vyla-api.hf.space/api/subtitles/movie/550'
).then(r => r.json());

// Apply to a video element
subtitles.forEach(sub => {
  const track = Object.assign(document.createElement('track'), {
    kind:  'subtitles',
    label: sub.label,
    src:   sub.file,
  });
  videoEl.appendChild(track);
});
```

## Check Provider Health

Before building, verify which providers are currently live:

```bash
curl https://missourimonster-vyla-api.hf.space/api/health
```

```json
{
  "status": "ok",
  "timestamp": "2025-01-01T00:00:00.000Z",
  "tmdb": true,
  "cache": 0,
  "probe_id": "155",
  "sources": {
    "provider-a": { "ok": true,  "ms": 1204 },
    "provider-b": { "ok": true,  "ms": 980  },
    "provider-c": { "ok": false, "ms": 435  }
  }
}
```

`status: "degraded"` means at least one provider is down — the rest still work and your fallback logic handles it automatically.

## Debug a Single Provider

Use `/api/test` to isolate a specific provider without running the full fanout:

```bash
# Test a provider for a movie
curl "https://missourimonster-vyla-api.hf.space/api/test/550?source=<provider-key>"

# Test a provider for a TV episode
curl "https://missourimonster-vyla-api.hf.space/api/test/1396?season=1&episode=1&source=<provider-key>"
```

<Tip>
  Available provider keys are returned by [`/api/health`](https://missourimonster-vyla-api.hf.space/api/health) under the `sources` field, or by fetching [`/api/sources`](https://missourimonster-vyla-api.hf.space/api/sources).
</Tip>
